0.priority_queue/map/unorder_map/set/vector

1.自定义排序

2.运算符重载

3.要注意的点： 一般自定义类型后要对运算符重载时，一般都是重载”<”运算符

而且，特别要注意，priority_queue 中若写为less，则重载”<”运算符后是维护大根堆！意思是top()迭代器指向的是”value”最大的元素；

#include<bits/stdc++.h>
using namespace std;

struct nd{
    int pos;
    int value;
    nd(int p,int v){
        this->pos=p;
        this->value=v;
    }
    //重载运算符"<"
    bool operator<(const nd& n)const{
        return value<n.value;
    }
};


//重写cmp
bool cmp(const pair<string,int> &p1,const pair<string,int> &p2){
    return p1.first>p2.first;
}



int main(){

//vector 自定义排序，binary_seaech,
//lower_bound/upper_bound的使用

vector<pair<string,int>> v;

v.push_back(pair<string,int>("ly",100));
v.push_back(pair<string,int>("dwz",1050));
v.push_back(pair<string,int>("qw",170));
v.push_back(pair<string,int>("jyx",723));

//按照value的升序排序
sort(v.begin(),v.end(),cmp);

//遍历输出 使用迭代器
for(vector<pair<string,int>>::iterator it=v.begin();it!=v.end();it++){
    cout<<(*it).first<<" "<<(*it).second<<endl;
}

//binary_search
//if(binary_search(strat,end,value)){...}

int a[4]={1,2,3,4};

/*lower_bound( begin,end,val )
lower_bound用以查找给定的数组区间中第一个大于或等于val的位置，不过如果传入的being、end是指针的话，返回的也是指针，想得到下标需将返回值减去start。注意考虑val可能大于区间中所有元素的情况，这是返回的是end的位置。
与此相似的还有upper_bound,不过查找的是第一个大于val的位置

*/

//优先队列，维护最大值（less），/最小值（greater），时间复杂度O(ln n)内完整编排
priority_queue<nd,vector<nd>,less<nd>> pq;
pq.push(nd(1,-1));
pq.push(nd(2,2));
pq.push(nd(3,100));
pq.push(nd(4,6));
cout<<pq.top().pos<<" "<<pq.top().value<<endl;


//set默认是升序
set<int> s;
s.insert(3);
s.insert(1);
s.insert(2);
set<nd> S;
S.insert(nd(1,3));
S.insert(nd(2,1));
S.insert(nd(3,9));
cout<<"基本数据类型的默认升序：(最小值)"<<*(s.begin())<<endl;
set<nd>::iterator it =S.begin();
cout<<upper_bound(S.begin(),S.end(),nd(4,1))->pos<<endl;
cout<<"自定义数据类型的重载(最小值)："<<it->pos<<" "<<it->value<<endl;
S.erase(S.begin());
cout<<S.begin()->pos<<endl;

//map
map<int,string> mapp;
mapp.insert(map<int,string>::value_type(10,"ly"));
mapp.insert(map<int,string>::value_type(2,"dwz"));
mapp.insert(map<int,string>::value_type(3,"jyx"));

map<int,string>::iterator itm=mapp.begin();
cout<<"map默认升序排列"<<itm->first<<" "<<itm->second<<endl;

unordered_map<int,string> my_un_map;


return 0;
}
