java中static介绍以及注意事项

package static_learn;

/**
 * 成员变量：
 * 1，静态成员变量，用static修饰，（属于类，在内存只加载一次，其实表征的
 * 是类中相较于实例成员变量来说，更具有一般化，即共有化的属性，相当于是归并中的归并）
 * 如实时在线人数
 * 一般使用类名.静态成员变量访问(A101)
 * 2，实例成员变量（存在于每个对象当中）
 * 它表征的是对象所共有，但，但是，但是更为个性化的属性（如在线人中各个人的id，年龄等）
 * 一般使用：对象.实例成员变量访问，(A102)
 *
 *
 * -------------------------------------------------------------
 * -------------------------------------------------------------
 *
 *
 * 成员方法（访问方式同成 员变量）
 * 1，静态成员方法，
 */
public class static_app {
    public static int num;
    private String name;

    public static void  getmax(int a,int b){
        System.out.println(a>b?a:b);
    }

    public void learn (String name){
        System.out.println(name+"niu");
    }
    public static void main(String[] args) {
        num--;
        System.out.println(num);//(A101)
        static_app t1 = new static_app();
        System.out.println(t1.name);//A(102)

        static_app.getmax(3,5);
        t1.learn("cjj");
    }
}
